import numpy as np
from scipy.optimize import minimize
import pandas as pd
import yfinance as yf

# Define assets (as a list, not a dictionary for optimization)
asset_list = ["AAPL", "NVDA", "MSFT", "GOOGL", "TSLA", "JPM", "XOM", "LLY", "REGN", "SPY", "GLD", "TLT"]

# Download or load your price data
# If you don't have adj_close yet, download it:
print("Downloading price data...")
adj_close = yf.download(asset_list, start="2024-01-01", end="2025-06-30", auto_adjust=True, progress=False)['Close']

# Calculate returns and annualized metrics
returns = adj_close.pct_change().dropna()
mean_returns = returns.mean() * 252  # Annualized mean returns
cov_matrix = returns.cov() * 252      # Annualized covariance matrix

n = len(asset_list)

def portfolio_volatility(weights):
    """Calculate portfolio volatility (standard deviation)"""
    return np.sqrt(np.dot(weights.T, np.dot(cov_matrix, weights)))

def portfolio_return(weights):
    """Calculate portfolio return"""
    return np.dot(weights, mean_returns)

def neg_sharpe(weights, risk_free_rate=0):
    """Negative Sharpe ratio (for minimization)"""
    ret = portfolio_return(weights)
    vol = portfolio_volatility(weights)
    return -(ret - risk_free_rate) / vol

# Constraints: weights sum to 1
constraints = ({'type': 'eq', 'fun': lambda x: np.sum(x) - 1})

# Bounds: individual weights between 0 and 0.15 (15% max per asset)
bounds = tuple((0, 0.15) for _ in range(n))

# Initial guess: equal weights
initial_weights = np.repeat(1/n, n)

# Optimize for maximum Sharpe ratio
opt = minimize(
    neg_sharpe,
    initial_weights,
    method='SLSQP',
    bounds=bounds,
    constraints=constraints
)

# Get optimal weights
optimal_weights = opt.x

# Calculate optimal portfolio metrics
optimal_return = portfolio_return(optimal_weights)
optimal_volatility = portfolio_volatility(optimal_weights)
optimal_sharpe = (optimal_return - 0) / optimal_volatility

# Display results
print("\n" + "=" * 60)
print("OPTIMAL PORTFOLIO (MAX SHARPE RATIO)")
print("=" * 60)

# Create DataFrame for weights
weights_df = pd.DataFrame({
    'Asset': asset_list,

print("\n📊 Optimal Weights:")
active_weights = weights_df[weights_df['Weight'] > 0.001]

print(f"\n📈 Portfolio Metrics:")

def marchenko_pastur_bounds(T,N,sigma=1):

    q = T/N

    lambda_max = sigma**2*(1+np.sqrt(1/q))**2
    lambda_min = sigma**2*(1-np.sqrt(1/q))**2

    return lambda_min,lambda_max


def rmt_denoise(corr):

    eigenvals, eigenvecs = np.linalg.eigh(corr)

    idx = eigenvals.argsort()[::-1]

    eigenvals = eigenvals[idx]
    eigenvecs = eigenvecs[:,idx]

    lambda_min,lambda_max = marchenko_pastur_bounds(T,N)

    noise = eigenvals <= lambda_max

    eigenvals_d = eigenvals.copy()

    eigenvals_d[noise] = eigenvals[noise].mean()

    corr_d = eigenvecs @ np.diag(eigenvals_d) @ eigenvecs.T

    corr_d = (corr_d + corr_d.T)/2

    np.fill_diagonal(corr_d,1)

    return corr_d,eigenvals,eigenvals_d,lambda_max

def get_cluster_var(cov,cluster):

    cov_slice = cov.loc[cluster,cluster]

    weights = 1/np.diag(cov_slice)

    weights /= weights.sum()

    variance = weights @ cov_slice @ weights

    return variance


def hrp_allocation(cov,link):

    order = leaves_list(link)

    ordered = cov.index[order]

    weights = pd.Series(1,index=ordered)

    clusters=[ordered]

    while len(clusters)>0:

        cluster = clusters.pop(0)

        if len(cluster)<=1:
            continue

        split = int(len(cluster)/2)

        c1 = cluster[:split]
        c2 = cluster[split:]

        var1 = get_cluster_var(cov,c1)
        var2 = get_cluster_var(cov,c2)

        alpha = 1 - var1/(var1+var2)

        weights[c1]*=alpha
        weights[c2]*=1-alpha

        clusters.append(c1)
        clusters.append(c2)

    return weights

# =============================================================================
# HIERARCHICAL RISK PARITY (HRP) PORTFOLIO
# =============================================================================

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy.cluster.hierarchy import linkage, leaves_list
from scipy.spatial.distance import squareform


# -----------------------------------------------------------------------------
# Compute cluster variance
# -----------------------------------------------------------------------------

def get_cluster_var(cov, cluster):

    cov_slice = cov.loc[cluster, cluster]

    ivp = 1 / np.diag(cov_slice)
    ivp = ivp / ivp.sum()

    cluster_var = ivp @ cov_slice @ ivp

    return cluster_var


# -----------------------------------------------------------------------------
# Recursive bisection (core HRP algorithm)
# -----------------------------------------------------------------------------

def recursive_bisection(cov, sorted_assets):

    weights = pd.Series(1, index=sorted_assets)

    clusters = [sorted_assets]

    while len(clusters) > 0:

        cluster = clusters.pop(0)

        if len(cluster) <= 1:
            continue

        split = int(len(cluster) / 2)

        cluster1 = cluster[:split]
        cluster2 = cluster[split:]

        var1 = get_cluster_var(cov, cluster1)
        var2 = get_cluster_var(cov, cluster2)

        alpha = 1 - var1 / (var1 + var2)

        weights[cluster1] *= alpha
        weights[cluster2] *= 1 - alpha

        clusters.append(cluster1)
        clusters.append(cluster2)

    return weights


# -----------------------------------------------------------------------------
# Main HRP function
# -----------------------------------------------------------------------------

def get_hrp_weights(cov_matrix, returns):

    corr = returns.corr()

    distance = np.sqrt(0.5 * (1 - corr))

    distance = distance.values
    np.fill_diagonal(distance, 0)

    link = linkage(squareform(distance), method="ward")

    sort_ix = leaves_list(link)

    sorted_assets = returns.columns[sort_ix]

    cov_sorted = cov_matrix.loc[sorted_assets, sorted_assets]

    weights = recursive_bisection(cov_sorted, list(sorted_assets))

    return weights


# =============================================================================
# COMPUTE HRP WEIGHTS
# =============================================================================

print(hrp_weights.sort_values(ascending=False))

def portfolio_performance_metrics(returns, risk_free_rate=0):

    TRADING_DAYS = 252

    # Annual return
    annual_return = returns.mean() * TRADING_DAYS

    # Annual volatility
    annual_volatility = returns.std() * np.sqrt(TRADING_DAYS)

    # Sharpe Ratio
    sharpe_ratio = (annual_return - risk_free_rate) / annual_volatility

    # -----------------------------
    # Sortino Ratio
    # -----------------------------
    downside_returns = returns[returns < 0]
    downside_volatility = downside_returns.std() * np.sqrt(TRADING_DAYS)

    sortino_ratio = (annual_return - risk_free_rate) / downside_volatility

    # -----------------------------
    # Max Drawdown
    # -----------------------------
    cumulative_returns = (1 + returns).cumprod()
    running_max = cumulative_returns.cummax()

    drawdown = cumulative_returns / running_max - 1
    max_drawdown = drawdown.min()

    # -----------------------------
    # Calmar Ratio
    # -----------------------------
    calmar_ratio = annual_return / abs(max_drawdown)

    # -----------------------------
    # Print results
    # -----------------------------
    print("\nPortfolio Performance Metrics")
    print("-" * 40)

    print(f"Annual Return      : {annual_return:.4f}")
    print(f"Annual Volatility  : {annual_volatility:.4f}")
    print(f"Sharpe Ratio       : {sharpe_ratio:.4f}")
    print(f"Sortino Ratio      : {sortino_ratio:.4f}")
    print(f"Max Drawdown       : {max_drawdown:.4f}")
    print(f"Calmar Ratio       : {calmar_ratio:.4f}")

    return {
        "Annual Return": annual_return,
        "Annual Volatility": annual_volatility,
        "Sharpe Ratio": sharpe_ratio,
        "Sortino Ratio": sortino_ratio,
        "Max Drawdown": max_drawdown,
        "Calmar Ratio": calmar_ratio,
        "Drawdown": drawdown
    }

# 2. PORTFOLIO STATISTICS

TRADING_DAYS = 252

mean_returns = returns.mean() * TRADING_DAYS
cov_matrix   = returns.cov()  * TRADING_DAYS

num_assets = len(assets)

# =============================================================================
# 3. MEAN-VARIANCE OPTIMIZATION
# =============================================================================

def portfolio_return(weights):
    return np.dot(weights, mean_returns)

def portfolio_volatility(weights, cov):
    return np.sqrt(np.dot(weights.T, np.dot(cov, weights)))

def negative_sharpe(weights, cov):
    ret = portfolio_return(weights)
    vol = portfolio_volatility(weights, cov)
    return -(ret / vol)

bounds = tuple((0,0.15) for _ in range(num_assets))

constraints = ({'type':'eq','fun':lambda x: np.sum(x)-1})

initial_weights = np.repeat(1/num_assets, num_assets)

print("Optimizing mean-variance portfolio...")

opt = minimize(
    negative_sharpe,
    initial_weights,
    args=(cov_matrix,),
    method='SLSQP',
    bounds=bounds,
    constraints=constraints
)

weights_original = pd.Series(opt.x, index=assets)

# =============================================================================
# 4. RANDOM MATRIX THEORY DENOISING
# =============================================================================

corr = returns.corr()

eigenvalues, eigenvectors = np.linalg.eigh(corr)

idx = eigenvalues.argsort()[::-1]
eigenvalues = eigenvalues[idx]
eigenvectors = eigenvectors[:,idx]

T = returns.shape[0]
N = returns.shape[1]

q = T / N

lambda_max = (1 + np.sqrt(1/q))**2

noise_mask = eigenvalues <= lambda_max

eigenvalues_denoised = eigenvalues.copy()
eigenvalues_denoised[noise_mask] = eigenvalues[noise_mask].mean()

corr_denoised = eigenvectors @ np.diag(eigenvalues_denoised) @ eigenvectors.T
corr_denoised = (corr_denoised + corr_denoised.T) / 2

np.fill_diagonal(corr_denoised,1)

corr_denoised = pd.DataFrame(corr_denoised, index=assets, columns=assets)

cov_denoised = corr_denoised * np.outer(returns.std(), returns.std())

# =============================================================================
# 5. DENOISED PORTFOLIO OPTIMIZATION
# =============================================================================

print("Optimizing denoised portfolio...")

opt2 = minimize(
    negative_sharpe,
    initial_weights,
    args=(cov_denoised,),
    method='SLSQP',
    bounds=bounds,
    constraints=constraints
)

weights_denoised = pd.Series(opt2.x, index=assets)

# =============================================================================
# 6. HIERARCHICAL RISK PARITY (HRP)
# =============================================================================

corr = returns.corr()

distance = np.sqrt(0.5 * (1 - corr))
distance = distance.values

np.fill_diagonal(distance,0)

link = linkage(squareform(distance), method='ward')

sort_ix = leaves_list(link)

sorted_assets = returns.columns[sort_ix]

cov_sorted = returns[sorted_assets].cov()

ivp = 1 / np.diag(cov_sorted)

weights_hrp = ivp / ivp.sum()

weights_hrp = pd.Series(weights_hrp, index=sorted_assets)

# =============================================================================

print("Running portfolio backtest...")

ret_original = returns @ weights_original

ret_denoised = returns @ weights_denoised

ret_hrp = returns[weights_hrp.index] @ weights_hrp

cum_original = (1 + ret_original).cumprod()

TRADING_DAYS = 252

def compute_risk_metrics(returns):
    """
    Compute key portfolio performance metrics:
    - Annualized return
    - Annualized volatility
    - Sharpe ratio
    - Sortino ratio
    - Max drawdown
    - Expected Shortfall (95%)
    """
    # Annualized return & volatility
    ann_return = returns.mean() * TRADING_DAYS
    ann_vol = returns.std() * np.sqrt(TRADING_DAYS)

    # Sharpe Ratio (risk-free rate = 0)
    sharpe = ann_return / ann_vol

    # Sortino Ratio (focus on downside risk)
    downside = returns[returns < 0]
    sortino = ann_return / (downside.std() * np.sqrt(TRADING_DAYS))

    # Max Drawdown
    cumulative = (1 + returns).cumprod()
    running_max = cumulative.cummax()
    drawdown = cumulative / running_max - 1
    max_dd = drawdown.min()

    # Expected Shortfall (95%)
    var95 = np.percentile(returns, 5)
    es95 = returns[returns <= var95].mean()

    return pd.Series({
        "Annual Return": ann_return,
        "Volatility": ann_vol,
        "Sharpe Ratio": sharpe,
        "Sortino Ratio": sortino,
        "Max Drawdown": max_dd,
        "Expected Shortfall": es95
    })

weights_combined = pd.Series(weights_combined, index=sorted_assets)
